    <script>
        let audioCtx = null;
        let isMusicPlaying = false;

        // --- Generador de Partículas de Fondo ---
        const pContainer = document.getElementById('particles');
        const emojis = ['❤️', '🧸', '✨', '🌸', '💕'];
        
        function makeParticles() {
            for(let i=0; i<15; i++) {
                setTimeout(() => {
                    const p = document.createElement('div');
                    p.classList.add('particle');
                    p.innerText = emojis[Math.floor(Math.random() * emojis.length)];
                    p.style.left = Math.random() * 100 + 'vw';
                    p.style.animationDuration = (Math.random() * 4 + 5) + 's';
                    pContainer.appendChild(p);
                    setTimeout(() => p.remove(), 9000);
                }, i * 400);
            }
        }
        setInterval(makeParticles, 8000);
        makeParticles();

        // --- Sistema de Navegación ---
        function goToPage(num) {
            document.querySelectorAll('.card').forEach(c => c.classList.add('hidden'));
            document.getElementById('page' + num).classList.remove('hidden');
        }

        function startExperience() {
            goToPage(2);
            if(!isMusicPlaying) {
                playMusicLoop();
                isMusicPlaying = true;
            }
        }

        // --- Comportamiento Botón "No" ---
        let noAttempts = 0;
        function escapeNo() {
            noAttempts++;
            if(noAttempts >= 3) {
                noAttempts = 0;
                const noBtn = document.getElementById('noBtn');
                noBtn.style.position = 'static';
                noBtn.style.transform = 'none';
                goToPage(3); // Castigo ositos
                return;
            }
            const noBtn = document.getElementById('noBtn');
            noBtn.style.position = 'fixed';
            const x = Math.random() * (window.innerWidth - 120);
            const y = Math.random() * (window.innerHeight - 50);
            noBtn.style.left = x + 'px';
            noBtn.style.top = y + 'px';
        }

        // --- Control de Acordeón ---
        function toggleAcc(header) {
            const content = header.nextElementSibling;
            if (content.style.maxHeight) {
                content.style.maxHeight = null;
                content.style.padding = "0 10px";
            } else {
                document.querySelectorAll('.acc-content').forEach(c => {
                    c.style.maxHeight = null;
                    c.style.padding = "0 10px";
                });
                content.style.maxHeight = content.scrollHeight + "px";
                content.style.padding = "10px";
            }
        }

        // --- Control de Slider de Perdón ---
        function checkSlider(val) {
            const status = document.getElementById('sliderStatus');

            if(val < 40) {

        // --- API de Audio Nativa: Notas de "Devuélveme a mi chica" ---
        function playMusicLoop() {
            try {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                
                // Transcripción aproximada en frecuencias [Frecuencia (Hz), Duración (ms)]
                const introMelody = [
                    [293.66, 200], [329.63, 200], [349.23, 200], [392.00, 400], [0, 100],
                    [392.00, 200], [349.23, 200], [329.63, 200], [293.66, 400], [0, 100],
                    [293.66, 200], [329.63, 200], [349.23, 200], [261.63, 400], [0, 100],
                    [349.23, 200], [329.63, 200], [293.66, 400]
                ];

                function playNode(freq, dur, startTime) {
                    if (freq === 0) return;
                    let osc = audioCtx.createOscillator();
                    let gain = audioCtx.createGain();
                    
                    osc.type = 'triangle'; // Tono retro/tierno
                    osc.frequency.setValueAtTime(freq, startTime);
                    
                    gain.gain.setValueAtTime(0.05, startTime);
                    gain.gain.exponentialRampToValueAtTime(0.0001, startTime + (dur/1000) - 0.02);
                    
                    osc.connect(gain);
                    gain.connect(audioCtx.destination);
                    osc.start(startTime);
                    osc.stop(startTime + (dur/1000));
                }

                function runLoop() {
                    let current = audioCtx.currentTime;
                    introMelody.forEach(note => {
                        playNode(note[0], note[1], current);
                        current += (note[1] / 1000) + 0.02;
                    });
                    setTimeout(runLoop, 6500); // Repetir bucle continuamente
                }
                
                runLoop();
            } catch (e) {
                console.log("Hardware de audio no inicializado correctamente.");
            }
        }
